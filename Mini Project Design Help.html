python
from PIL import Image, ImageDraw, ImageFont, ImageFilter
import random, string, os

def generate_captcha():
    captcha_text = ''.join(random.choices(string.ascii_uppercase + string.digits, k=5))
    width, height = 150, 60
    image = Image.new('RGB', (width, height), (255, 255, 255))

    font = ImageFont.truetype("arial.ttf", 40)
    draw = ImageDraw.Draw(image)

    # Add background noise
    for _ in range(500):
        x = random.randint(0, width)
        y = random.randint(0, height)
        draw.point((x, y), fill=(random.randint(0,255), random.randint(0,255), random.randint(0,255)))

    # Draw text
    for i, char in enumerate(captcha_text):
        draw.text((10 + i*25, random.randint(5,15)), char, font=font, fill=(0, 0, 0))

    # Save image
    path = f"static/captcha_images/{captcha_text}.png"
    image.save(path)
    return captcha_text, path

ðŸŒ Step 4: Web Interface with Flask

âœï¸ app.py

python
from flask import Flask, render_template, request, session, redirect
from captcha_generator import generate_captcha
import time, os, csv

app = Flask(__name__)
app.secret_key = 'your_secret_key'

@app.route('/', methods=['GET', 'POST'])
def captcha():
    if request.method == 'POST':
        user_input = request.form['captcha']
        correct = session['captcha_text']
        start_time = float(request.form['start_time'])
        solve_time = time.time() - start_time

        is_bot = False
        if solve_time < 1.5:
            is_bot = True
        if request.form.get('honeypot'):
            is_bot = True

        # Log the result
        with open('logs.csv', 'a') as file:
            writer = csv.writer(file)
            writer.writerow([time.strftime('%Y-%m-%d %H:%M:%S'), request.remote_addr, solve_time, is_bot])

        if user_input == correct and not is_bot:
            return "CAPTCHA Solved Successfully!"
        else:
            return "CAPTCHA Failed or Bot Detected!"

    captcha_text, path = generate_captcha()
    session['captcha_text'] = captcha_text
    return render_template('captcha_form.html', img_path=path, start_time=time.time())

if __name__ == "__main__":
    if not os.path.exists('static/captcha_images'):
        os.makedirs('static/captcha_images')
    app.run(debug=True)
